```js react-live
function LikeButton() {
    const [likes, increaseLikes] = React.useState(0)

    return (
        <>
            <p class="likes">{likes} likes</p>
            <button
                onClick={() => increaseLikes(likes + 1)} />
        </>
    )
}

```
