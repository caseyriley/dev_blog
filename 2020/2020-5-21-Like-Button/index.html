Like Button

2021 May 21st

This is a basic like button I made for playing with code :)
Try changing "likes + 1" to "likes + 3".
Then click on the heart.
Have fun with inline styles if your feeling it.

function LikeButton() {
  const [likes, increaseLikes] = React.useState(0);


  return (
    <>
      <p style={{ fontSize: "22px", color: "purple" }}>{likes} likes</p>


      <button
        style={{
          height: "50px",
          width: "50px",
          borderRadius: "10px",
          borderColor: "purple",
          fontSize: "30px",
        }}
        onClick={() => increaseLikes(likes + 1)}
      >
        <span role={"img"} aria-label={"gold heart"}>

        </span>
      </button>
    </>
  );
}
